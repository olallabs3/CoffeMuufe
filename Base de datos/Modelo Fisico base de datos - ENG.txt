 CREATE TABLE PAY_METHOD (
ID_PAY_METHOD NUMBER(15) CONSTRAINT PK_PAY_METHOD PRIMARY KEY,
CASH NUMBER(5,2),  
BANK_CARD NUMBER(16)   
);

CREATE TABLE CLIENT (
USER_ID VARCHAR2(30) CONSTRAINT PK_CLIENT PRIMARY KEY,  //depende de la longitud que se le quiera dar
FIRST_NAME VARCHAR2(30),
LAST_NAME VARCHAR2(30),
PASSWORD CHAR(30),   
EMAIL CHAR(30),
NACIONALILTY VARCHAR(30),
PHONE_NUMBER NUMBER(10)   
);


CREATE TABLE ADDRESS(
ADDRESS_ID NUMBER(20) CONSTRAINT PK_ADDRESS PRIMARY KEY,
STREET VARCHAR2(30),
NUMBER NUMBER(30),
FLAT CHAR(10),
POSTAL_CODE NUMBER(10)
);


CREATE TABLE PRODUCT (
ID_PRODUCT NUMBER(20) CONSTRAINT PK_PRODUCT PRIMARY KEY,
NAME VARCHAR2(30),
INGREDIENTS VARCHAR2(30),
PRICE NUMBER(5,2)    
);

CREATE TABLE CATEGORY (
CATEGORY_ID NUMBER(20) CONSTRAINT PK_CATEGORY PRIMARY KEY,
CATEGORY_NAME VARCHAR2(20)
);

CREATE TABLE ORDERS(
PRODUCT_ORDER_ID(20) CONSTRAINT PK_ORDERS PRIMARY KEY,  
PRICE NUMBER(5,2),  
NOTES CHAR(100),   
PRODUCT VARCHAR2(20),
DATE DATE,   
PAY_METHOD_ID NUMBER(15),
DIRECTION_ID NUMBER(20),
);

CREATE TABLE ORDER_LINES (
ORDER_LINES_ID NUMBER(30) CONSTRAINT PK_ORDER_LINES PRIMARY KEY,
ORDER_ID NUMBER(20),
PRODUCT_ID NUMBER(20),
QUANTITY NUMBER(20)
);
